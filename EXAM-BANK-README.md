```sql
-- Get all tricky questions
SELECT * FROM exam_bank 
WHERE course_id = 'pcs125' AND is_tricky = 1;

-- Get questions by topic
SELECT * FROM exam_bank 
WHERE course_id = 'pcs125' AND topic_area = 'simple_harmonic_motion';

-- Get questions by difficulty
SELECT * FROM exam_bank 
WHERE course_id = 'pcs125' AND difficulty = 3 
ORDER BY exam_year DESC;
```
